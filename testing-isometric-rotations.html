# Define factor for how much to lighten/darken the sides of the cuboids
f <- 0.25

# Function takes 3D xyz and returns 2D xy through rotation matrices m1a, m1b and m2 (defined later)
iso <- Vectorize(function(x,y,z) m2 %*% ((m1a %*% m1b) %*% matrix(c(x, y, z))))

m1a <- 
  matrix(c(1, 0, 0,

         ncol = 3)

m1b <-

unit_cube_df <-
  tribble(

    "a",  0,  0,  0,
    "a",  0,  1,  0,
    "a",  1,  1,  0,
    "a",  1,  0,  0,

    "b",  0,  0,  -1,
    "b",  0,  1,  -1,
    "b",  1,  1,  -1,
    "b",  1,  0,  -1,

    "c",  0,  0,  0,
    "c",  0,  0,  -1,
    "c",  0,  1,  -1,
    "c",  0,  1,  0,

    "d",  1,  0,  0,
    "d",  1,  0,  -1,

    "e",  0,  0,  0,
    "e",  1,  0,  0,
    "e",  1,  0,  -1,
    "e",  0,  0,  -1,

    "f",  0,  1,  0,

    "f",  0,  1,  -1
  ) %>% 
  filter(g %in% c("b", "c", "f"))

uc <- unit_cube_df %>% mutate(iso(ux, uy, uz) %>% t() %>% as_tibble(.name_repair = ~c("x", "y", "z")))

ggplot()+
  geom_polygon(data = uc, aes(x, y, group = g, fill = g))+
  geom_polygon(data = uc %>% rename(g2=g), aes(x, y, group = g2), col = 1, fill=NA)+
  coord_equal()+
  facet_wrap(~g)

img_df <- 
  expand.grid(z = nrow(m):1,
              x = 1:ncol(m)) %>% 
  mutate(y = v, col = cols, block_id = row_number()) %>% 
  merge(unit_cube_df, all = TRUE) %>% 
  as_tibble() %>% 
  mutate(ux = ux+x, uy = uy*y, uz = uz+z)

img_df %>% 
  mutate(iso(ux, uy, uz)[1:2,] %>% t() %>% as_tibble(.name_repair = ~c("px", "py"))) %>%
  mutate(col2rgb(col) %>% rgb2hsv() %>% t() %>% as_tibble(),
         v2 = case_when(g == "f" ~ v, # top
                        g == "b" ~ v*(1-f), # right
                        g == "c" ~ v*(1-(2*f))), #left
         col2 = hsv(h, s, v2),
         my_group = paste0(g, block_id)) %>% 
  arrange(desc(x), desc(z)) %>%
  mutate(my_group = fct_inorder(my_group)) %>%
  ggplot(aes(px, py, fill=I(col2), group=my_group))+
  geom_polygon(alpha=1, col=NA, size = 0.25)+
  coord_equal()+
  theme_void()
